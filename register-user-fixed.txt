  <script type="module">
    import bcrypt from 'https://cdn.jsdelivr.net/npm/bcryptjs@2.4.3/+esm';
    const el = (id) => document.getElementById(id);
    const log = (t) => el('msg').textContent = t;

    const sleep = (ms) => new Promise(r => setTimeout(r, ms));

    function buildHeaders(anonKey, fnToken) {
      const h = { 'Content-Type': 'application/json' };
      if (fnToken) h['x-edsa-token'] = fnToken;
      // For REST insert we also need anon auth headers:
      h['apikey'] = anonKey;
      h['Authorization'] = `Bearer ${anonKey}`;
      return h;
    }

    function validateInputs() {
      const email = el('email').value.trim();
      if (!/^[^\\s@]+@[^\\s@]+\\.[^\\s@]+$/.test(email)) throw new Error('Please enter a valid email.');
      const idnum = el('id_number').value.trim();
      if (!/^[0-9]{6,}$/.test(idnum)) throw new Error('Please enter a valid ID number (6+ digits).');
      if (!el('role').value) throw new Error('Please select a role.');
      if (!el('branch').value) throw new Error('Please select a branch.');
    }

    async function tryCallEmail(fnUrl, body, headers) {
      const res = await fetch(fnUrl, { method: 'POST', mode: 'cors', headers, body: JSON.stringify(body) });
      const text = await res.text().catch(()=>''); // capture full body
      return { ok: res.ok, status: res.status, statusText: res.statusText, text };
    }

    async function sendEmailFlow(anonKey, fnToken) {
      const base = el('sb_url').value.trim().replace(/\\/$/,''); // strip trailing slash
      const fn = el('fn_name').value.trim() || 'send-user-email-public';

      const payload = {
        to: el('email').value.trim().toLowerCase(),
        id_number: el('id_number').value.trim(),
        full_name: el('full_name').value.trim(),
        branch: el('branch').value.trim() || 'Erase Debt SA - ADMIN'
      };

      const headers = { 'Content-Type': 'application/json' };
      if (fnToken) headers['x-edsa-token'] = fnToken;

      // Try both function endpoints (some setups prefer functions.supabase.co domain)
      const urls = [
        `${base}/functions/v1/${fn}`,
        base.replace('.supabase.co', '.functions.supabase.co') + `/functions/v1/${fn}`
      ];

      let results = [];
      for (const u of urls) {
        try {
          const r = await tryCallEmail(u, payload, headers);
          results.push({ url: u, ...r });
          if (r.ok) break; // stop on success
        } catch (e) {
          results.push({ url: u, ok: false, status: 0, statusText: 'Network/Fetch error', text: String(e) });
        }
        await sleep(100);
      }

      return results;
    }

    async function insertUser(anonKey) {
      const base = el('sb_url').value.trim().replace(/\\/$/,''); // strip slash
      const body = {
        full_name: el('full_name').value.trim(),
        email: el('email').value.trim().toLowerCase(),
        id_number: el('id_number').value.trim(),
        password_hash: await bcrypt.hash(Math.random().toString(36).slice(2), 10),
        role: el('role').value,
        branch: el('branch').value,
        status: el('status').value,
        is_active: el('status').value === 'active'
      };

      const res = await fetch(`${base}/rest/v1/edsa_users`, {
        method: 'POST',
        mode: 'cors',
        headers: {
          'apikey': anonKey,
          'Authorization': `Bearer ${anonKey}`,
          'Content-Type': 'application/json',
          'Prefer': 'return=representation'
        },
        body: JSON.stringify(body)
      });
      const text = await res.text().catch(()=>'');
      return { ok: res.ok, status: res.status, statusText: res.statusText, text };
    }

    function renderResults(title, arr) {
      const lines = [title];
      for (const r of arr) {
        lines.push(`→ ${r.url || ''}`.trim());
        lines.push(`   HTTP ${r.status} ${r.statusText}`);
        if (r.text) {
          lines.push(`   Body: ${r.text.slice(0, 2000)}`);
        }
      }
      log(lines.join('\\n'));
    }

    el('btnTestOnly').addEventListener('click', async () => {
      try {
        const anon = el('sb_anon').value.trim();
        const token = el('fn_token').value.trim();
        if (!token) throw new Error('Paste your Edge Function token first.');
        const results = await sendEmailFlow(anon, token);
        renderResults('Test Email Function Results:', results);
      } catch (e) {
        log('❌ ' + e.message);
      }
    });

    el('btnCreate').addEventListener('click', async () => {
      try {
        validateInputs();
        const anon = el('sb_anon').value.trim();
        if (!anon) throw new Error('Paste your Supabase anon key first.');
        const token = el('fn_token').value.trim();
        if (!token) throw new Error('Paste your Edge Function token first.');

        log('⏳ Inserting user...');
        const ins = await insertUser(anon);
        if (!ins.ok) {
          log(`❌ Failed to insert user. HTTP ${ins.status} ${ins.statusText}\n${ins.text}`);
          return;
        }
        log('✅ User inserted. Now sending email...');
        const results = await sendEmailFlow(anon, token);
        renderResults('Email Function Results:', results);
      } catch (e) {
        log('❌ ' + e.message);
      }
    });
  </script>
